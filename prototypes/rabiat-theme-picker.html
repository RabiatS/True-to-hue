/* ================================
   RABIAT THEME v1.0
   Personal Design System
   ================================ */

:root {
  /* ── BLACKS ── */
  --r-void:     #0A0A0A;  /* base background */
  --r-obsidian: #141414;  /* surface */
  --r-shadow:   #1E1E1E;  /* card background */
  --r-midnight: #2A2A2A;  /* elevated surface */

  /* ── WHITES ── */
  --r-bone:   #F5F5F0;    /* primary text / white */
  --r-pearl:  #E8E8E3;    /* secondary white */
  --r-ash:    #A0A09A;    /* muted text */
  --r-slate:  #5A5A55;    /* subdued */

  /* ── REDS ── */
  --r-bloodline: #8B0000; /* primary red */
  --r-marrow:    #6B0000; /* deep red */
  --r-signal:    #B50000; /* bright red */
  --r-ember:     #3D0000; /* red tint background */

  /* ── TEAL (YOUR PICK) ── */
  --r-teal:       #2DD4BF; /* primary — Neon Lagoon */
  --r-teal-dark:  #1A9B8A; /* deep current */
  --r-teal-light: #7EEADC; /* vapor */
  --r-teal-bg:    #031A18;  /* depth (tint bg) */
}
